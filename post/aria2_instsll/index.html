Aria2 一键安装管理脚本 增强版

2020-05-18
centos shell
次阅读

本人非常排斥迅雷,因为他会偷您的流量,所以所有电脑从来不装迅雷软件,由于是运维所以手上倒是有很多linux服务器,此次用aria2替代

#安装http
wget -N git.io/aria2.sh && chmod +x aria2.sh
#最好是把系统更新关了
sed "s|yum update|#yum update|" aria2.sh
cat /root/.aria2/aria2.conf |grep sec


#手动安装
yum install epel-release #安装EPEL源
yum install aria2 -y
#配置Aria2
mkdir .aria2
cd .aria2
touch aria2.session
touch aria2.log

#配置文件
vi /root/.aria2/aria2.conf
#文件保存路径设置，请手动更改
dir=/opt/aria2

disk-cache=32M
file-allocation=none
continue=true
max-concurrent-downloads=10
max-connection-per-server=5
min-split-size=10M
split=20
disable-ipv6=true
input-file=/root/.aria2/aria2.session
save-session=/root/.aria2/aria2.session

## RPC相关设置 ##
# 启用RPC, 默认:false
enable-rpc=true
# 允许所有来源, 默认:false
rpc-allow-origin-all=true
# 允许非外部访问, 默认:false
rpc-listen-all=true
# 事件轮询方式, 取值:[epoll, kqueue, port, poll, select], 不同系统默认值不同
#event-poll=select
# RPC监听端口, 端口被占用时可以修改, 默认:6800
rpc-listen-port=6800
# 设置的RPC授权令牌，在设置AriaNg时需要用到，请手动更改
rpc-secret=1a97fea302ce243f51d4

follow-torrent=true
listen-port=6881-6999
enable-dht=true
enable-peer-exchange=true
peer-id-prefix=-TR2770-
user-agent=Transmission/2.77
seed-ratio=0.1
bt-seed-unverified=true
bt-save-metadata=false


#启动
aria2c --conf-path=/root/.aria2/aria2.conf -D

#开机自启
vi /etc/rc.d/rc.local
aria2c --conf-path=/root/.aria2/aria2.conf -D &
chmod +x /etc/rc.d/rc.local



#安装AriaNG
wget https://github.com/mayswind/AriaNg/releases/download/1.1.3/AriaNg-1.1.3.zip
unzip AriaNg-1.1.3.zip
